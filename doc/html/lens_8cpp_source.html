123 // Compute lensing potential via convolution in Fourier space (this requires kappa to be initialized)
124 void lensT::compute_psi_from_kappa()
125 {
132  int orig_w = kappa.cols;
133  int orig_h = kappa.rows;
134  int opt_2w = cv::getOptimalDFTSize(2*orig_w);
135  int opt_2h = cv::getOptimalDFTSize(2*orig_h);
136  cv::copyMakeBorder(kappa, kappa, 0, opt_2h-orig_h, 0, opt_2w-orig_w, cv::BORDER_CONSTANT);
137 
138  // Create the Green's function kernel G
139  Mat G = Mat::zeros(opt_2h, opt_2w, CV_64FC1);
140  Mat G_hat, kappa_hat, product;
141  fill_green_fct(G);
142 
143  // Apply DFT to G and kappa, multiply and backward transform their results to obtain psi
144  cv::dft(G, G_hat, cv::DFT_REAL_OUTPUT);
145  cv::dft(kappa, kappa_hat, cv::DFT_REAL_OUTPUT);
146  cv::mulSpectrums(kappa_hat, G_hat, product, 0, false);
147  cv::idft(product, psi, cv::DFT_SCALE);
148 
149  // Crop all maps back to the original size of kappa
150  cv::Rect crop_region(0, 0, orig_w, orig_h);
151  psi = psi(crop_region);
152  kappa = kappa(crop_region);
153 }

156 void lensT::compute_derivatives_from_psi()
